💡 Resumo do Funcionamento

⚡
A Importância dos Webhooks: Em ambos os fluxos, seu banco de dados nunca é atualizado direto na tela do usuário. O Gateway envia uma requisição automática (Webhook) para o seu servidor assim que o pagamento é liquidado. Isso previne fraudes e garante a liberação do produto mesmo se o usuário fechar a aba do navegador antes da confirmação.

Exemplos de implementação (Node.js/Next.js) para criar a lógica no seu Backend.

const axios = require('axios');

// 🚀 CRIAÇÃO DE COBRANÇA AVULSA PARA BOOST (ASAAS)
async function criarBoostAvulso(userId, postAnuncioId, valor) {
    const url = 'https://api.asaas.com/v3/payments';
    
    const payload = {
        "customer": "cus_9238491823", // ID do cliente (Luthier/Professor)
        "billingType": "PIX",          // Cobrança via Pix
        "value": valor,               // Ex: R$ 15.00
        "dueDate": "2026-06-10",       // Vence hoje/amanhã
        
        // Armazenamos dados de referência para ler no Webhook depois
        "externalReference": `boost_${postAnuncioId}_${userId}`
    };

    const response = await axios.post(url, payload, {
        headers: { "access_token": "SEU_TOKEN_ASAAS" }
    });
    
    // Retorna a chave Copia e Cola e o QR Code gerado
    return response.data;
}
const Stripe = require('stripe');
const stripe = Stripe('sua_chave_secreta_stripe');

// 👑 CRIAÇÃO DE ASSINATURA MENSAL DE TIER (STRIPE BILLING)
async function criarAssinaturaTier(customerId, priceIdToken) {
    // Cria a assinatura do plano mensal selecionado
    const subscription = await stripe.subscriptions.create({
        "customer": customerId,
        "items": [{
            "price": priceIdToken, // ID do Preço do Tier (ex: Pro - R$ 49.90/mês)
        }],
        "payment_behavior": "default_incomplete",
        "payment_settings": { "save_default_payment_method": "on_subscription" },
        "expand": ["latest_invoice.payment_intent"],
    });

    return {
        subscriptionId: subscription.id,
        clientSecret: subscription.latest_invoice.payment_intent.client_secret
    };
}
